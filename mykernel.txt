__kernel void fadd(__global float* in1, __global float* in2, __global float* out)
{
    int idx = get_global_id(0);
    
    out[idx] = in1[idx] * 3.174f + in2[idx] +
               in1[idx] * in2[idx] + in1[idx]+ 
               in1[idx] - in2[idx]*3.141516 + 
	       in1[idx] * 3.141516 + in2[idx]; 
}
